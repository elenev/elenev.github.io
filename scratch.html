from bs4 import BeautifulSoup

# Example code to import and parse an HTML file using BeautifulSoup

# Assuming the HTML file is named 'example.html' and is located in the same directory as this script
file_path = 'CV.html'

# Open the HTML file
with open(file_path, 'r') as file:
    # Use BeautifulSoup to parse the HTML content
    soup = BeautifulSoup(file, 'html.parser')
# Initialize the presentations list and temporary variables
presentations = []
current_title = None
current_talks = []

def store_talks(title, talks):
    """Store the talks in the presentations list"""
    presentations.append({
        "Title": title,
        "Talks": talks
    })

sub = soup.find("span", string="Presentations")

# Iterate over the table rows and construct the JSON structure
for row in sub.find_all_next('tr'):
    # Check for title row
    strong_tag = row.find('strong')
    if strong_tag:
        if current_title:  # If there is a current title, append the talks before starting a new title
            store_talks(current_title, current_talks)
            current_talks = []
        current_title = strong_tag.text
        continue

    first_td = row.find('td').find_next('td')

    if first_td.text == '':
        continue
    elif first_td.text == 'Conference discussions':
        store_talks(current_title, current_talks)
        break

    # Split by first comma
    if str(first_td.text).count(',') > 1:
        venue, location = first_td.text.split(',', 1)
    else:
        venue = first_td.text
        location = ''

    second_td = first_td.find_next('td')
    type = second_td.text

    month = second_td.find_next('td').text

    # Append the talk to the current talks list
    current_talks.append({
        "Venue": venue.strip(),
        "Location": location.strip(),
        "Type": type.strip(),
        "Month": month.strip()
    })
